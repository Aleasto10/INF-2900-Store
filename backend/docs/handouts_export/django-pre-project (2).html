Model tests: 

class BookModelTest(TestCase):
    def test_empty_title_invalid(self):
        emptybook = Book(title='', author='me', isbn='123-456-6790-1')
        with self.assertRaises(ValidationError):
            emptybook.full_clean()

    def test_duplicate_isbn_invalid(self):
        dup_isbn = '123-456-6790-1'
        Book.objects.create(title='one', author='me', isbn=dup_isbn)
        with self.assertRaises(ValidationError):
            dup_isbn_book = Book(title='two', author='you', isbn=dup_isbn)
            dup_isbn_book.validate_unique()

View tests: 

class BookViewTest(TestCase):
    def test_use_book_template(self):
        b1 = Book.objects.create(title='b1', author='john', isbn='123')
        response = self.client.get(f'/books/{b1.id}/')
        self.assertTemplateUsed(response, 'book.html')

    def test_context_has_the_book(self):
        b1 = Book.objects.create(title='b1', author='john', isbn='123')
        response = self.client.get(f'/books/{b1.id}/')
        self.assertEqual(response.context['book'], b1)

    def test_redirect_when_book_does_not_exist(self):
        response = self.client.get('/books/99/')
        self.assertRedirects(response, '/books/')

Acceptance test: 

class VisitAndCreateTest(StaticLiveServerTestCase):

    def setUp(self):
        self.browser = webdriver.Firefox()

    def tearDown(self):
        self.browser.quit()

    def test_visit_and_create(self):
        self.browser.get(f'{self.live_server_url}/books/new')
        title, author, isbn = 'agile', 'me', '321'
        inputbox = self.browser.find_element(By.ID, 'id_title')
        inputbox.send_keys(title)
        inputbox.send_keys(Keys.ENTER)
        inputbox = self.browser.find_element(By.ID, 'id_author')
        inputbox.send_keys(author)
        inputbox.send_keys(Keys.ENTER)
        inputbox = self.browser.find_element(By.ID, 'id_isbn')
        inputbox.send_keys(isbd)
        inputbox.send_keys(Keys.ENTER)
        time.sleep(2)
        inputbox = self.browser.find_element(By.ID, 'id_title')
        self.assertEqual(inputbox.get_attribute('value'), title)
        inputbox = self.browser.find_element(By.ID, 'id_author')
        self.assertEqual(inputbox.get_attribute('value'), author)
        inputbox = self.browser.find_element(By.ID, 'id_isbn')
        self.assertEqual(inputbox.get_attribute('value'), isbn)
